# Cell 3 — Protocol Generators (P_phys schedule & W' logic)

def generate_protocol_power_schedule(minutes: int,
                                     protocol_type: str,
                                     target_intensity: str,
                                     CP_W: float) -> np.ndarray:
    """
    Create a target 'desired' power schedule (the physiological cap before W' depletion),
    in Watts, for the given protocol.
    """
    sched = np.zeros(minutes, dtype=float)

    if protocol_type == "steady":
        if target_intensity == "at_CP":
            sched[:] = CP_W
        elif target_intensity == "CP_minus_10pct":
            sched[:] = 0.90 * CP_W
        elif target_intensity == "CP_plus_10pct":
            sched[:] = 1.10 * CP_W
        else:
            sched[:] = CP_W

    elif protocol_type == "intervals":
        # Alternate blocks (2 min @ 1.1*CP, 2 min @ 0.9*CP) across the ride
        t = 0
        idx = 0
        while t < minutes:
            dur, mult = INTERVALS_BLOCKS[idx % len(INTERVALS_BLOCKS)]
            end = min(minutes, t + dur)
            sched[t:end] = mult * CP_W
            t = end
            idx += 1

        # If user passes target_intensity, bias towards harder or easier:
        if target_intensity == "at_CP":
            sched = np.where(sched > CP_W, 1.05*CP_W, 0.95*CP_W)
        elif target_intensity == "CP_plus_10pct":
            sched = np.where(sched > CP_W, 1.15*CP_W, 1.00*CP_W)

    elif protocol_type == "TT5min":
        # Build steady at CP, then a 5-min TT at 1.25*CP placed around minute 20 (if ride long enough)
        sched[:] = CP_W
        tt_start = min(20, max(0, minutes - 5)//2)
        tt_end = min(minutes, tt_start + 5)
        sched[tt_start:tt_end] = TT5_MULTIPLIER * CP_W
        if target_intensity == "CP_plus_10pct":
            sched[tt_start:tt_end] = 1.30 * CP_W

    else:
        sched[:] = CP_W

    return sched

def apply_wprime_balance(target_sched: np.ndarray, CP_W: float, Wprime_kJ: float, dt_min: float) -> np.ndarray:
    """
    Apply W' depletion/recovery to cap feasible power above/below CP.
    Simplified Skiba model: when P>CP consume W', when P<CP recover towards W'max at a modest rate.
    """
    minutes = len(target_sched)
    feasible = np.zeros_like(target_sched)
    Wprime = Wprime_kJ * 1000.0  # J
    Wprime_max = Wprime
    tau_recovery_min = 10.0      # simple recovery time constant (minutes)
    for t in range(minutes):
        P_des = target_sched[t]
        if P_des > CP_W:
            # Can we sustain above CP this minute?
            over = P_des - CP_W  # Watts
            need_J = over * 60.0  # J for this minute
            if need_J <= Wprime:
                feasible[t] = P_des
                Wprime -= need_J
            else:
                # W' exhausted: cap at CP + remaining/60
                feasible[t] = CP_W + (Wprime / 60.0)
                Wprime = 0.0
        else:
            feasible[t] = P_des
            # Recover a fraction of the deficit toward W' max
            rec = (Wprime_max - Wprime) * (dt_min / tau_recovery_min)
            Wprime += rec
            Wprime = min(Wprime, Wprime_max)
    return feasible

# Cell 4 — Absorption/Metabolism (Minute-Resolution)

def simulate_absorption_metabolism(E_ethanol_kJ: float,
                                   E_carb_kJ: float,
                                   t_half_GE_min: float,
                                   absorption_lag_min: int,
                                   Vmax_kJ_per_min: float,
                                   Km_kJ: float,
                                   gross_efficiency: float,
                                   ride_duration_min: int,
                                   ingest_duration_min: int,
                                   ingest_to_ride_min: int) -> Dict[str, np.ndarray]:
    """
    Returns dict with:
      EtOH_plasma_kJ[t], carb_gut_kJ[t], EtOH_hep_rate_kJmin[t], carb_rate_kJmin[t],
      usable_rate_kJmin[t] (after efficiency), BrAC_proxy[t]
    Timeline: from minute 0 = ride start. Drinking happens before ride (negative minutes).
    """
    # Build a pre-ride timeline to handle absorption before ride starts
    pre_minutes = ingest_to_ride_min
    total_minutes = pre_minutes + ride_duration_min

    # Distribute ethanol & carbs into "gut pools" across the ingestion window
    gut_EtOH = np.zeros(total_minutes)
    gut_Carb = np.zeros(total_minutes)

    # Ingestion occurs from t = -(ingest_duration_min) to 0 relative to drink end,
    # but we index 0..pre_minutes-1 before ride: pour into earliest pre-ride minutes.
    # Approximate: uniform ingestion over ingest_duration_min right after scenario start offset.
    start_ingest = max(0, pre_minutes - ingest_duration_min)
    end_ingest = pre_minutes
    if end_ingest > start_ingest:
        per_min_etoh = E_ethanol_kJ / (end_ingest - start_ingest)
        per_min_carb = E_carb_kJ / (end_ingest - start_ingest)
        gut_EtOH[start_ingest:end_ingest] = per_min_etoh
        gut_Carb[start_ingest:end_ingest] = per_min_carb

    # Gastric emptying exponential rate
    k_GE = math.log(2.0) / max(1.0, t_half_GE_min)

    # Pools
    gut_pool_EtOH = 0.0
    gut_pool_Carb = 0.0
    plasma_EtOH = 0.0  # effective "plasma-accessible" ethanol energy pool (kJ), surrogate

    EtOH_plasma_kJ = np.zeros(total_minutes)
    carb_gut_kJ = np.zeros(total_minutes)
    EtOH_hep_rate = np.zeros(total_minutes)  # kJ/min metabolized via hepatic pathway
    carb_rate = np.zeros(total_minutes)      # kJ/min available from carbs (no saturation)
    usable_rate = np.zeros(total_minutes)    # kJ/min after gross efficiency
    brac_proxy = np.zeros(total_minutes)

    # Simulate minute by minute
    for t in range(total_minutes):
        # Add freshly ingested energy into gut pools this minute
        gut_pool_EtOH += gut_EtOH[t]
        gut_pool_Carb += gut_Carb[t]

        # Gastric emptying to "plasma-accessible" (for EtOH) and "ready" (for carbs)
        # Apply absorption lag: nothing exits gut for lag minutes after ingestion start
        if t >= start_ingest + absorption_lag_min:
            # Empty a fraction from gut pools
            empty_etoh = gut_pool_EtOH * (1.0 - math.exp(-k_GE * 1.0))
            empty_carb = gut_pool_Carb * (1.0 - math.exp(-k_GE * 1.0))
            gut_pool_EtOH -= empty_etoh
            gut_pool_Carb -= empty_carb
            plasma_EtOH += empty_etoh
            # Carbs assume direct availability after emptying
            carb_ready = empty_carb
        else:
            empty_etoh = 0.0
            carb_ready = 0.0

        # Hepatic processing of ethanol (Michaelis–Menten)
        # rate = Vmax * pool / (Km + pool)
        hep_rate = 0.0
        if plasma_EtOH > 0:
            hep_rate = Vmax_kJ_per_min * (plasma_EtOH / (Km_kJ + plasma_EtOH))
            hep_rate = min(hep_rate, plasma_EtOH)  # can't process more than pool left
            plasma_EtOH -= hep_rate

        # Carbohydrate availability this minute (kJ/min):
        # send at most what is "ready"; the rest remains in gut_pool_Carb (already handled)
        carb_avail = carb_ready

        # Combine, apply gross efficiency to get usable mechanical-rate bound
        mech_kJmin = gross_efficiency * (hep_rate + carb_avail)

        # BrAC proxy (surrogate from plasma pool scaled by body mass to be added later)
        EtOH_plasma_kJ[t] = plasma_EtOH
        carb_gut_kJ[t] = gut_pool_Carb
        EtOH_hep_rate[t] = hep_rate
        carb_rate[t] = carb_avail
        usable_rate[t] = mech_kJmin
        brac_proxy[t] = plasma_EtOH  # raw; scaled later by mass/penalty_alpha

    # Slice to ride window
    ride_slice = slice(pre_minutes, total_minutes)
    out = {
        "EtOH_plasma_kJ": EtOH_plasma_kJ[ride_slice],
        "carb_gut_kJ":    carb_gut_kJ[ride_slice],
        "EtOH_hep_rate":  EtOH_hep_rate[ride_slice],
        "carb_rate":      carb_rate[ride_slice],
        "usable_rate_kJmin": usable_rate[ride_slice],
        "brac_proxy_raw": brac_proxy[ride_slice],
    }
    return out

# Cell 5 — Penalty & Final Power Composition

def compute_penalty(brac_raw_series: np.ndarray,
                    mass_kg: float,
                    penalty_alpha_per_mgdl: float) -> np.ndarray:
    """
    Convert raw 'plasma energy pool' surrogate into a mg/dL-like proxy and compute penalty.
    We scale by body mass and clamp. Penalty = 1 - alpha * BrAC_proxy, clamped [0.7, 1.0].
    """
    # naive scaling: more mass → lower concentration proxy
    scaled = (brac_raw_series / max(1.0, mass_kg)) * BRAC_SCALE
    scaled = np.clip(scaled, *BRAC_CLAMP)
    penalty = 1.0 - penalty_alpha_per_mgdl * scaled
    return np.clip(penalty, 0.7, 1.0)


def compose_final_power(usable_rate_kJmin: np.ndarray,
                        feasible_p_phys_W: np.ndarray,
                        penalty: np.ndarray) -> Tuple[np.ndarray, float]:
    """
    Convert usable kJ/min -> Watts, cap by physiologic schedule, apply penalty.
    Returns P(t) watts and fraction of energy utilized.
    """
    supply_W = usable_rate_kJmin * (1000.0 / 60.0)
    base = np.minimum(supply_W, feasible_p_phys_W)
    P = base * penalty

    # Fraction energy utilized: (integral of base power) / (integral of supply power)
    supply_work = supply_W.sum() * 60.0  # J
    base_work = base.sum() * 60.0        # J
    frac = 0.0 if supply_work <= 0 else (base_work / supply_work)
    return P, frac


# --- Helper energy conversion utilities (previously missing) ---

def ethanol_energy_kJ(abv_pct: float, dose_ml: float) -> float:
    """
    Convert beverage ABV (%) and volume (ml) to ethanol chemical energy in kJ.
    Uses density constant RHO_ETOH_G_PER_ML and energy per gram ETHANOL_KJ_PER_G.
    """
    # mass of ethanol in grams
    mass_g = RHO_ETOH_G_PER_ML * dose_ml * (abv_pct / 100.0)
    return mass_g * ETHANOL_KJ_PER_G


def carb_energy_kJ(sugar_g: float) -> float:
    """Simple conversion: grams of sugar -> kJ using CARB_KJ_PER_G."""
    return sugar_g * CARB_KJ_PER_G


def blended_label_energy_kJ(label_kcal: float) -> float:
    """Convert label kcal to kJ (used if scenarios include a labelled kcal value)."""
    return label_kcal * 4.184


def blended_total_energy_kJ(E_etoh_kJ: float, E_carb_kJ: float, E_label_kJ: float) -> float:
    """Combine different energy sources into a single kJ total (simple sum for now)."""
    return float(E_etoh_kJ + E_carb_kJ + E_label_kJ)

# Cell 6 — Single-Scenario Simulation

def simulate_scenario(row: pd.Series) -> Dict[str, float]:
    """
    Simulate one scenario (one row). Returns summary metrics + time series (for optional saving).
    Required columns: as per your schema.
    """
    # Inputs
    mass_kg = float(row["mass_kg"])
    abv_pct = float(row["abv_pct"])
    dose_ml = float(row["dose_ml"])
    label_kcal = float(row["label_kcal"])
    sugar_g = float(row["sugar_g"])
    ingest_duration_min = int(row["ingest_duration_min"])
    ingest_to_ride_min = int(row["ingest_to_ride_min"])

    CP_W = float(row["CP_W"])
    Wprime_kJ = float(row["Wprime_kJ"])
    gross_eff = float(row["gross_efficiency"])

    t_half_GE_min = int(row["t_half_GE_min"])
    absorption_lag_min = int(row["absorption_lag_min"])
    Vmax_kJ_per_min = float(row["Vmax_kJ_per_min"])
    Km_kJ = float(row["Km_kJ"])
    penalty_alpha = float(row["penalty_alpha_per_mgdl"])

    ride_duration_min = int(row["ride_duration_min"])
    protocol_type = str(row["protocol_type"])
    target_intensity = str(row["target_intensity"])

    # Energy accounting
    E_etoh = ethanol_energy_kJ(abv_pct, dose_ml)
    E_carb = carb_energy_kJ(sugar_g)
    E_label = blended_label_energy_kJ(label_kcal)
    E_total = blended_total_energy_kJ(E_etoh, E_carb, E_label)

    # Absorption/metabolism to minute-resolved supply
    abs_res = simulate_absorption_metabolism(
        E_ethanol_kJ=E_etoh,
        E_carb_kJ=E_carb,
        t_half_GE_min=t_half_GE_min,
        absorption_lag_min=absorption_lag_min,
        Vmax_kJ_per_min=Vmax_kJ_per_min,
        Km_kJ=Km_kJ,
        gross_efficiency=gross_eff,
        ride_duration_min=ride_duration_min,
        ingest_duration_min=ingest_duration_min,
        ingest_to_ride_min=ingest_to_ride_min,
    )

    usable_rate = abs_res["usable_rate_kJmin"]
    brac_raw = abs_res["brac_proxy_raw"]

    # Physiological demand/cap
    sched = generate_protocol_power_schedule(
        minutes=ride_duration_min,
        protocol_type=protocol_type,
        target_intensity=target_intensity,
        CP_W=CP_W
    )
    feasible = apply_wprime_balance(sched, CP_W, Wprime_kJ, dt_min=1.0)

    # Penalty vs BrAC proxy
    penalty = compute_penalty(brac_raw, mass_kg=mass_kg, penalty_alpha_per_mgdl=penalty_alpha)

    # Final realized power
    P_t, frac_used = compose_final_power(usable_rate, feasible, penalty)

    # Summaries
    P_peak_W = float(np.max(P_t))
    HP_peak = P_peak_W / WATT_PER_HP
    t_peak_min = int(np.argmax(P_t))
    P_mean_W = float(np.mean(P_t))
    work_kJ = float(np.sum(P_t) * 60.0 / 1000.0)  # W*60 /1000 = kJ per minute step
    min_penalty = float(np.min(penalty))

    # Return both summaries and (optionally) time series
    out = dict(
        P_peak_W=round(P_peak_W, 3),
        HP_peak=round(HP_peak, 5),
        t_peak_min=int(t_peak_min),
        P_mean_W=round(P_mean_W, 3),
        work_kJ=round(work_kJ, 3),
        frac_energy_utilized=round(float(frac_used), 4),
        min_penalty=round(min_penalty, 4),
        _timeseries=dict(
            P_t=P_t,
            feasible=feasible,
            penalty=penalty,
            usable_rate_kJmin=usable_rate,
            EtOH_plasma_kJ=abs_res["EtOH_plasma_kJ"],
            EtOH_hep_rate=abs_res["EtOH_hep_rate"],
            carb_rate=abs_res["carb_rate"]
        )
    )
    return out
